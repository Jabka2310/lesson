# Yazi with directory change
function yy() {
    local temp_file=$(mktemp)
    command yazi --cwd-file "$temp_file" "$@"
    if [[ -f "$temp_file" ]]; then
        local dir=$(cat "$temp_file")
        if [[ -n "$dir" && -d "$dir" ]]; then
            cd "$dir"
        fi
        rm "$temp_file"
    fi
}


export EDITOR=nvim
export VISUAL=nvim
alias vim=nvim
